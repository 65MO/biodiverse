32 
33  next if $i == 0;
34 
35  $indices{"HIER_ARAT$i\_$j"} = {description => "Ratio of A scores, (HIER_A$i / HIER_A$j)", lumper => 0,};
36  $indices{"HIER_BRAT$i\_$j"} = {description => "Ratio of B scores, (HIER_B$i / HIER_B$j)", lumper => 0,};
37  $indices{"HIER_CRAT$i\_$j"} = {description => "Ratio of C scores, (HIER_C$i / HIER_C$j)", lumper => 0,};
38  $indices{"HIER_ASUMRAT$i\_$j"} = {description => "1 - Ratio of shared label sample counts, (HIER_ASUM$i / HIER_ASUM$j)",
39  cluster => 'NO_CACHE_ABC', # value is true, but allows a caveat
40  };
41  }
42  }
43 
44  #my $levels = ($column_count - 1);
45  my $desc = <<"END_H_DESC"
46 Analyse the diversity of labels using their hierarchical levels.
47 The A, B and C scores are the same as in the Label Counts analysis (calc_label_counts)
48 but calculated for each hierarchical level, e.g. for three axes one could have
49 A0 as the Family level, A1 for the Family:Genus level,
50 and A2 for the Family:Genus:Species level.
51 The number of indices generated depends on how many axes are used in the labels.
52 In this case there are $column_count. Axes are numbered from zero
53 as the highest level in the hierarchy, so level 0 is the top level
54 of the hierarchy.
55 END_H_DESC

76 sub calc_hierarchical_label_ratios {
77  my $self = shift;
78  my %args = @_;
79 
80  my $hierarchy = $args{BD_HIERARCHY};
81 
82  my %results;
83 
84  my @sum_a_keys;
85 
86  foreach my $i (0 .. $#$hierarchy) {
87  no warnings qw /numeric uninitialized/; # divide by zero will return undef
88  # undef will be treated as zero
89 
90  #my $bd = $hierarchy->[$i];
91  my $xx = $self->calc_abc3 (
92  element_list1 => $args{element_list1},
93  element_list2 => $args{element_list2},
94  );
95 
96  $results{'HIER_A' . $i} = $$xx{A};
97  $results{'HIER_B' . $i} = $$xx{B};
98  $results{'HIER_C' . $i} = $$xx{C};
99 
100  my $xx_a_keys = $self -> get_shared_hash_keys (
101  lists => [
102  $xx->{label_hash1},
103  $xx->{label_hash2}
104  ]
105  );
106  $sum_a_keys[$i] = 0;
107  foreach my $key (keys %$xx_a_keys) {
108  $sum_a_keys[$i] += $xx->{label_hash_all}{$key};
109  }
110 
111  $results{'HIER_ASUM' . $i} = $sum_a_keys[$i];
112 
113  next if $i == 0;
114 
115  my $j = $i - 1;
116 
117  $results{"HIER_ARAT$i\_$j"} = eval {$xx->{A} / $results{'HIER_A' . $j}};
118  $results{"HIER_BRAT$i\_$j"} = eval {$xx->{B} / $results{'HIER_B' . $j}};
119  $results{"HIER_CRAT$i\_$j"} = eval {$xx->{C} / $results{'HIER_C' . $j}};
120  # if the denominator is zero then we have no overlap
121  # between label lists -- make it a result of 1
122  $results{"HIER_ASUMRAT$i\_$j"}
123  = 1 - eval {$sum_a_keys[$i] / $sum_a_keys[$j]};
124  }
125 
126  return wantarray ? %results : \%results;

133 sub get_basedatas_by_label_hierarchy {
134  my $self = shift;
135  my %args = @_;

137  #my $progress_bar = $args{progress};

139 
140  my $bd = $self->get_basedata_ref;
141 
142  my $labels_ref = $bd->get_labels_ref;
143 
144  my $label_max_index = $bd->get_label_column_count - 1;
145 
146  my %results = (BD_HIERARCHY => []);
147  return wantarray ? %results : \%results if ! $label_max_index;
148 
149  # generate a set of new basedata objects
150  foreach my $i (0 .. $label_max_index) {
151  # just to allow greater abstraction, eg we may extend the basedata types
152  my $cell_sizes = $bd->get_param('CELL_SIZES');
153  push @{$results{BD_HIERARCHY}}, blessed ($bd)->new (CELL_SIZES => [@$cell_sizes]);
154  }
155 
156  my $targets = $results{BD_HIERARCHY};
157 
158  my $progress_count = 0;
159  my $to_do = $labels_ref -> get_element_count;
160 
161  foreach my $label ($labels_ref -> get_element_list) {
162  my @element_array = $labels_ref -> get_element_name_as_array (
163  element => $label,
164  );
165  my %groups = $bd->get_groups_with_label_as_hash (label => $label);
166 
167  $progress_count ++;
168  $progress_bar -> update (
169  "Building hierarchical basedata\n"
170  . "for label columns 0 to $label_max_index\n"
171  . "($progress_count / $to_do)",
172  $progress_count / $to_do
173  );
174 
175  foreach my $group (keys %groups) {
176  # get the count of the original
177  my $count = $groups{$group};
178 
179  foreach my $i (0 .. $label_max_index) {
180  # get the new label from the slice
181  my $new_label = $bd -> list2csv (
182  list => [@element_array[0..$i]]
183  );
184 
185  # now add this new label/group pair to the new basedata
186  $targets->[$i] -> add_element (
187  label => $new_label,
188  group => $group,
189  count => $count,
190  );
191  }
192  }
193  }
194 
195  return wantarray ? %results : \%results;
196 }
