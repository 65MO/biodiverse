1111 sub swap_to_reach_targets {
1112  my $self = shift;

1114

1124  || $args{basedata_ref};
1125  my $progress_bar = Biodiverse::Progress->new();
1126

1128  # shunt labels in and out of groups until we're happy
1129

1131  # Select an unassigned label.
1132  # Find a group that does not contain it.
1133  # Swap this label with one of the labels in the group if it is full.
1134  # Repeat until we have no more to assign or all groups are full

1137  + (scalar keys %unfilled_groups);

1141  }

1143  my $swap_count = 0;
1144  my $last_filled = $EMPTY_STRING;
1145

1147  BY_UNFILLED_GP:
1148  while (scalar keys %unfilled_groups) {
1149 
1150  my $target_label_count = $cloned_bd->get_label_count;
1151  my $target_group_count = $cloned_bd->get_group_count;
1152

1164  (scalar keys %unfilled_groups),
1165  (scalar keys %filled_groups),
1166  $target_label_count,

1188  my @labels = sort $cloned_bd->get_labels;

1212  # shuffling lists many times.
