137 sub generateWidget {
138  my $param = $_[0];
139 
140  my $type = $param->{type};
141 
142  my @valid_choices = qw {
143  file
144  integer
145  float
146  boolean
147  choice
148  spatial_params
149  };
150  my %valid_choices_hash;
151  @valid_choices_hash{@valid_choices} = (1) x scalar @valid_choices;
152 
153  croak "Unsupported parameter type $type\n"
154  if ! exists $valid_choices_hash{$type};
155 
156  my $sub_name = 'generate_' . $type;
157 
158  my @results = eval "$sub_name (\@_)";
159  croak "Unsupported parameter type $type\n" if $EVAL_ERROR;
160 
161  return @results;
162 }
