106 # Creates a TreeView model of available calculations
107 sub makeCalculationsModel {
108  my $base_ref = shift;
109  my $output_ref = shift;
110 
111  # Try to get a list of analyses that should be checked
112  my $checkRef;
113  if ($output_ref) { # the latter are for backwards compatibility
114  $checkRef = $output_ref->get_param('CALCULATIONS_REQUESTED')
115  or $output_ref->get_param('ANALYSES_REQUESTED')
116  or $output_ref->get_param('ANALYSES_RAN');
117  }
118  if (! defined $checkRef) {
119  $checkRef = [$EMPTY_STRING] ; # trap other cases
120  }
121 
122  #print "[Spatial tab] Analyses ran - " . join (" ", @$checkRef) . "\n" if scalar @$checkRef;
123 
124  my @treestore_args = (
125  'Glib::String', # Name
126  'Glib::String', # Index
127  'Glib::String', # Description
128  'Glib::String', # Function
129  'Glib::Boolean', # Checked?
130  'Glib::Boolean', # Grayed? (if some children checked but not all)
131  'Glib::Boolean', # Show checkbox? (no if description "second line")
132  );
133 
134  my $model = Gtk2::TreeStore->new( @treestore_args );
135 
136  #my $analysis_caller_ref = defined $output_ref ? $output_ref : $base_ref;
137  #my %calculations = $analysis_caller_ref->get_calculations;
138  my $indices = Biodiverse::Indices->new(BASEDATA_REF => $base_ref);
139  my %calculations = $indices->get_calculations;
140 
141  my $name_wrapper = Text::Wrapper->new(columns => $NAME_COLUMN_WIDTH);
142  my $desc_wrapper = Text::Wrapper->new(columns => $DESC_COLUMN_WIDTH);
143 
144  # Add the type row (eg: taxonomic, matrix)
145  foreach my $type (sort keys %calculations) {
146  my $type_iter = $model->append(undef);
147  $model->set(
148  $type_iter,
149  MODEL_NAME_COL, $type,
150  MODEL_CHECKED_COL, 0,
151  MODEL_GRAYED_COL, 0,
152  MODEL_SHOW_CHECKBOX, 1
153  );
154 
155  my %calc_metadata;
156  foreach my $func (@{$calculations{$type}}) {
157  my %info = $indices->get_args (sub => $func);
158  # If name unspecified then use the function name less the calc_
159  my $name = $func;
160  $name =~ s/^calc_//;
161  $name = $info{name} || $name;
162  $info{func} = $func;
163  $calc_metadata{$name} = \%info;
164  }
165 
166  # Add each analysis-function (eg: Jaccard, Endemism) row
167  CALCULATION_NAME:
168  foreach my $name (sort keys %calc_metadata) {
169  my %info = %{$calc_metadata{$name}};
170 
171  # If description goes over one line,
172  # we put the first line here and the next into a new row
173  # but combine all the other lines into a single line and then rewrap.
174  my $desc = $info{description} || $EMPTY_STRING;
175  my ($d1, $dRest) = split(/\n/, $desc, 2);
176 
177  # Rewrap the descriptions.
178  # Perl works on aliases so this will change the original strings.
179  foreach my $string ($d1, $dRest) {
180  next if ! defined $string;
181  $string =~ s/\A\n//;
182  $string =~ s/\n\z//;
183  $string =~ s/\n/ /g;
184  $string = $desc_wrapper->wrap($string);
185  $string =~ s/\n\z//;
186  }
187 
188  my @index_data = ();
189  push @index_data, ['Indices:', $EMPTY_STRING];
190 
191  # now loop over the indices to get their descriptions
192  foreach my $index (sort keys %{$info{indices}}) {
193  my $description = $info{indices}{$index}{description} || $EMPTY_STRING;
194  $description =~ s/\n//g; # strip newlines
195  $description = $desc_wrapper->wrap($description);
196  $description =~ s/\n\z//; # strip trailing newlines
197  #my @string = split ("\n", $description);
198 
199  push @index_data, [$index, $description];
200  }
201 
202  my $func = $info{func};
203 
204  # Should it be checked? (yes, if it was on previous time)
205  my $checked = member_of($func, $checkRef);
206 
207  # Add to model
208  my $func_iter = $model->append($type_iter);
209  $model->set(
210  $func_iter,
211  MODEL_NAME_COL, $name,
212  MODEL_CHECKED_COL, $checked,
213  MODEL_GRAYED_COL, 0,
214  MODEL_DESCRIPTION_COL, $d1,
215  MODEL_FUNCTION_COL, $func,
216  MODEL_SHOW_CHECKBOX, 1
217  );
218 
219  # Add multiline-description row
220  if ($dRest) {
221  my $desc_iter = $model->append($func_iter);
222  $model->set(
223  $desc_iter,
224  MODEL_DESCRIPTION_COL, $dRest,
225  MODEL_SHOW_CHECKBOX, 0,
226  );
227  }
228 
229  # add index descriptions
230  foreach my $index_pair (@index_data) {
231  my $index_iter = $model->append($func_iter);
232  $model->set(
233  $index_iter,
234  MODEL_INDEX_COL, $index_pair->[0],
235  MODEL_DESCRIPTION_COL, $index_pair->[1],
236  MODEL_SHOW_CHECKBOX, 0,
237  );
238  }
239  }
240 
241  # Check it if all calculations are checked
242  updateTypeCheckbox($model, $type_iter);
243  }
244 
245  return $model;
246 }

248 # Returns whether an element is in some array-ref
249 sub member_of {
250  my ($elem, $ref) = @_;
251 
252  # sometimes we are passed a hash
253  return exists $ref->{$elem}
254  if (reftype ($ref) eq 'HASH');
255 
256  foreach my $member (@$ref) {
257  return 1 if $elem eq $member;
258  }
259 
260  return 0;
261 }

263 sub getCalculationsToRun {
264  my $model = shift;
265  my @toRun;
266 
267  # Retrieve all calculations with a check mark
268  my $type_iter = $model->get_iter_first();
269  while ($type_iter) {
270 
271  my $child_iter = $model->iter_nth_child($type_iter, 0);
272 
273  while ($child_iter) {
274  my ($checked) = $model->get($child_iter, MODEL_CHECKED_COL);
275  if ($checked) {
276  my ($func) = $model->get($child_iter, MODEL_FUNCTION_COL);
277  push (@toRun, $func);
278  }
279  $child_iter = $model->iter_next($child_iter);
280  }
281 
282  $type_iter = $model->iter_next($type_iter);
283  }
284  return @toRun;
285 }

287 # Called to set the analysis type checkbox depending on whether all children are set
288 sub updateTypeCheckbox {
289  my $model = shift;
290  my $iter_top = shift || return;
291 
292  my $child_iter = $model->iter_nth_child($iter_top, 0);
293  my $prevchecked = undef;
294  my $check = 0;
295  my $gray = 0; # will be set to 1 if have both checked & unchecked children
296 
297  # Look for any unchecked children
298  while ($child_iter) {
299  ($check) = $model->get($child_iter, MODEL_CHECKED_COL);
300  if (defined $prevchecked && $prevchecked != $check) {
301  $gray = 1;
302  last;
303  }
304  $prevchecked = $check;
305  $child_iter = $model->iter_next($child_iter);
306  }
307 
308  $check = 0 if $gray == 1;
309 
310  $model->set($iter_top, MODEL_GRAYED_COL, $gray);
311  $model->set($iter_top, MODEL_CHECKED_COL, $check);
312 }

314 # Called when the user clicks on a checkbox
315 sub onCalculationToggled {
316  my $model = shift;
317  my $path = shift;
318 
319  my $iter = $model->get_iter_from_string($path);
320 
321  # Flip state
322  my $state = $model->get($iter, MODEL_CHECKED_COL);
323  my $grayed = $model->get($iter, MODEL_GRAYED_COL);
324 
325  if ($grayed == 1) {
326  $state = 1; # if clicked on a grayed checkbox - make it selected
327  }
328  else {
329  $state = not $state;
330  }
331  $model->set($iter, MODEL_CHECKED_COL, $state);
332  $model->set($iter, MODEL_GRAYED_COL, 0);
333 
334  # Apply state to all child nodes
335  my $child_iter = $model->iter_nth_child($iter, 0);
336  while ($child_iter) {
337  $model->set($child_iter, MODEL_CHECKED_COL, $state);
338  $child_iter = $model->iter_next($child_iter);
339  }
340 
341  # update state of any parent
342  updateTypeCheckbox($model, $model->iter_parent($iter) );
343 
344  return;
345 }
