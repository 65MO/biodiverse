152 sub variance {
153  my $self = shift; ##Myself
154 
155  my $count = $self->count();
156 
157  return undef if !$count;
158 
159  return 0 if $count == 1;
160 
161  if (!defined($self->_variance())) {
162  my $variance = ($self->sumsq()- $count * $self->mean()**2);
163 
164  # Sometimes due to rounding errors we get a number below 0.
165  # This makes sure this is handled as gracefully as possible.
166  #
167  # See:
168  #
169  # https://rt.cpan.org/Public/Bug/Display.html?id=46026
170  if ($variance < 0) {
171  $variance = 0;
172  }
173  else {
174  # Commented code is a left-over from early version.
175  # Assume it was to allow for biased method variance,
176  # but docs do not list it so assume it is unnecessary
177  # Actually, it is trapped by the $count == 1 condition above, so shouldn't be needed
178  #my $div = scalar @_ ? 0 : 1;
179  #$variance /= $count - $div;
180  $variance /= $count - 1;
181  }
182 
183  $self->_variance($variance);
184 
185  # return now to avoid sub re-entry (and therefore time when many objects are used)
186  return $variance;
187  }

313  my $sd = $self->standard_deviation();
314 
315  my $skew;
316 
317  # skip if insufficient records
318  if ( $sd && $n > 2) {
319 
320  my $mean = $self->mean();
321 
322  my @tmp = List::MoreUtils::apply { $_ = (($_ - $mean) / $sd) ** 3 } $self->get_data();
323  my $sum_pow3 = List::Util::sum @tmp;
324 
325  my $correction = $n / ( ($n-1) * ($n-2) );
326

328  }

352 
353  my $correction1 = ( $n * ($n+1) ) / ( ($n-1) * ($n-2) * ($n-3) );
354  my $correction2 = ( 3 * ($n-1) ** 2) / ( ($n-2) * ($n-3) );
355 
356  $kurt = ( $correction1 * $sum_pow4 ) - $correction2;
357  }

385 =head1 DESCRIPTION
386 
387 Basic descriptive statistics.
388 Everything from module Statistics::Descriptive::Full but with a
389 different percentile algorithm
390 (the original can be called using percentile_RFC2330).
391 
392 The median method also returns undef when there are no records.
