18 =head1
19 
20 Implements the popup dialogs shown when cells on the grid
21 are clicked. They can be reused - meaning that instead of making a new dialog,
22 an existing one is "overwritten".
23 
24 =head2
25 
26 The dialog is given a hash of SOURCE_NAME => $function
27 When a source is selected, $function is called and passed a $popup parameter.
28 Other parameters can be passed by making $function a closure, eg:
29 
30  $sources->{LABELS} = sub { showLabels(@_, $basedata_ref, $element); } ;
31  ($popup) is given as part of "@_"
32 
33 $popup is of type Biodiverse::GUI::PopupObject and has methods
34  setListModel - shows the given GTK model as the output
35  setValueColumn - puts the given model column onto the list's second column
36  This list's first column is the first model column
37 
38 
39 =cut

86 sub showPopup {
87  my $element = shift;
88  my $sources_ref = shift;
89  my $default_source = shift;
90  my $dlgxml;
91 
92  # If already showing a dialog, close it
93  if (exists $g_dialogs{$element}) {
94  closeDialog($element);
95  }
96  else {
97  if (defined $g_reuseDlg) {
98  $dlgxml = $g_reuseDlg;
99  delete $g_dialogs{$g_reuseElement};
100  #print "[Popup] Reusing dialog which was for $g_reuseElement\n";
101  }
102  else {
103  #print "[Popup] Making new labels dialog for $element\n";
104  $dlgxml = makeDialog();
105  }
106 
107  $g_dialogs{$element} = $dlgxml;
108  loadDialog($dlgxml, $element, $sources_ref, $default_source);
109  }

152 sub loadDialog {
153  my $dlgxml = shift;
154  my $element = shift;
155  my $sources_ref = shift;
156  my $default_source = shift;
157 
158  #print Data::Dumper::Dumper($neighbours);
159  #print Data::Dumper::Dumper(%g_dialogs);
160 
161  # Create pseudo-object hash to hold everything together
162  my $popup = {};
163  bless $popup, 'Biodiverse::GUI::PopupObject';
164 
165  $popup->{list} = $dlgxml->get_widget('lstData');
166  $popup->{element} = $element;
167  $popup->{sources_ref} = $sources_ref;
168 
169  # Create model of available sources
170  my $sources_model = makeSourcesModel($sources_ref);
171  #print "[Popup] Made source model\n";
172 
173  # Set up the combobox
174  my $combo = $dlgxml->get_widget('comboSources');
175  $combo->set_model($sources_model);
176 
177  my $selected_source = findSelectedSource($sources_model, $g_selectedSource) # first use user-selected
178  || findSelectedSource($sources_model, $default_source) # then try default source
179  || $sources_model->get_iter_first; # use first one otherwise
180  $combo->set_active_iter($selected_source);
181 
182  # Set title
183  $g_dialogs{$element}->get_widget(DLG_NAME)->set_title("Data for $element");
184 
185  # Load first thing
186  onSourceChanged($combo, $popup);
187 
188  # Disconnect signals (dialog might be being reused)
189  $dlgxml->get_widget('comboSources')->signal_handlers_disconnect_by_func(\&onSourceChanged);
190  $dlgxml->get_widget('btnClose')->signal_handlers_disconnect_by_func(\&closeDialog);
191  $dlgxml->get_widget(DLG_NAME)->signal_handlers_disconnect_by_func(\&closeDialog);
192  $dlgxml->get_widget('btnCloseAll')->signal_handlers_disconnect_by_func(\&onCloseAll);
193  $dlgxml->get_widget('btnCopy')->signal_handlers_disconnect_by_func(\&onCopy);
194  $dlgxml->get_widget('chkReuse')->signal_handlers_disconnect_by_func(\&onReuseToggled);
195 
196  # Connect signals
197  $dlgxml->get_widget('comboSources')->signal_connect(changed => \&onSourceChanged, $popup);
198  $dlgxml->get_widget('btnClose')->signal_connect_swapped(clicked => \&closeDialog, $element);
199  $dlgxml->get_widget(DLG_NAME)->signal_connect_swapped(delete_event => \&closeDialog, $element);
200  $dlgxml->get_widget('btnCloseAll')->signal_connect_swapped(clicked => \&onCloseAll);
201  $dlgxml->get_widget('btnCopy')->signal_connect_swapped(clicked => \&onCopy, $popup);
202 
203  # Set to last re-use state
204  #print "[Popup] last reuse = $g_lastReuse\n";
205  $dlgxml->get_widget('chkReuse')->set_active($g_lastReuse);
206  $dlgxml->get_widget('chkReuse')->signal_connect(toggled => \&onReuseToggled, [$element, $dlgxml]);
207  onReuseToggled($dlgxml->get_widget('chkReuse'), [$element, $dlgxml]);
208 }

371 sub clipboard_get_func {
372  my $clipboard = shift;
373  my $selection = shift;
374  my $datatype = shift;
375  my $popup = shift;
376 
377  #print "[Popup] Clipboard data request (type $datatype)\n";
378 
379  my $element = $popup->{element};
380  my $list = $popup->{list};
381  my $listname = $popup->{listname};
382  my $model = $list->get_model();
383  my $text;
384 
385  if (! $model) {
386  my $gui = Biodiverse::GUI::GUIManager->instance;
387  my $e = "Unable to paste data.\nPopup has been closed so link with source data is lost\n";
388  $gui->report_error($e);
389  return;
390  }
391 
392  # Start off with the "element" (ie: cell coordinates)
393  if ($datatype == TYPE_HTML) {
394  $text =<<'END_HTML_HEADER'
395  <!DOCTYPE html PUBLIC "-//W3C//DTD XHTML 1.0 Transitional//EN"
396  "http://www.w3.org/TR/xhtml1/DTD/xhtml1-transitional.dtd">
397  <html xmlns="http://www.w3.org/1999/xhtml" xml:lang="en" lang="en">
398 
399  <head>
400  <meta http-equiv="Content-Type" content="text/html; charset=utf-8" />
401  </head>
402 
403  <body>
404 
405  <table>
406 END_HTML_HEADER
407 ;
408  $text .= "<tr><td>$listname</td><td>$element</td></tr>";
409  }
410  else {
411  $text = "$listname\t$element\n";
412  }
413 
414  # Generate the text
415  my $iter;
416  eval {
417  $iter = $model->get_iter_first();
418  };
419  if ($EVAL_ERROR) {
420  my $gui = Biodiverse::GUI::GUIManager->instance;
421  $gui->report_error($EVAL_ERROR);
422  return;
423  }
424 
425  while ($iter) {
426  my $name = $model->get($iter, 0);
427  my $value = '';
428 
429  if ($popup->{value_column}) {
430  $value = $model->get($iter, $popup->{value_column});
431  }
432 
433  if ($datatype == TYPE_TEXT) {
434  $text .= "$name\t$value\n";
435  }
436  elsif ($datatype == TYPE_HTML) {
437  $text .= "<tr><td>$name</td><td>$value</td></tr>\n";
438  }
439  $iter = $model->iter_next($iter);
440  }
441 
442  if ($datatype == TYPE_HTML) {
443  $text .= "</table></body></html>\n";
444  }
445 
446  # Give the data..
447  print "[Popup] Sending data for $element to clipboard\n";
448 
449  if ($datatype == TYPE_HTML) {
450  my $atom = Gtk2::Gdk::Atom->intern('text/html');
451  $selection->set($atom, 8, $text);
452  }
453  elsif ($datatype == TYPE_TEXT) {
454  $selection->set_text($text);
455  }
456 
457  return;
458 }
