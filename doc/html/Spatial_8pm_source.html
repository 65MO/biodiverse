704  # get everything

707  }

729  delete @tmp{@$excl};
730  $nbr_list[$i] = [keys %tmp];
731  }

733 
734  # don't use the index if there are no search blocks
735  my $sp_index_i = defined $search_blocks_ref->[$i]
736  ? $sp_index
737  : undef;
738 
739  my %args_for_nbr_list = (
740  element => $element,
741  spatial_params => $spatial_params_ref->[$i],
742  index => $sp_index_i,
743  index_offsets => $search_blocks_ref->[$i],
744  );
745  my $exclude_list = [@exclude, @$elements_to_exclude];
746

748  my $progr = Biodiverse::Progress->new(text => 'Neighbour comparisons for first always_same condition');
749  my $tmp = $bd->get_neighbours (
750  %args_for_nbr_list,
751  progress => $progr,
752  );
753  $progr = undef;
754  my $cached_arr = $self->get_cached_value('NBRS_FROM_ALWAYS_SAME');
755  if (!$cached_arr) {
756  $cached_arr = [];
757  $self->set_cached_value(NBRS_FROM_ALWAYS_SAME => $cached_arr);
758  }
759 
760  $cached_arr->[$i] = $tmp;
761  my %tmp2 = %$tmp;
762  delete @tmp2{@$exclude_list};
763  $nbr_list[$i] = [keys %tmp2];
764  }
765  else {
766  # go search
767  $nbr_list[$i] = $bd->get_neighbours_as_array (
768  %args_for_nbr_list,
769  exclude_list => $exclude_list,
770  );
771  }
772 
773  }
774 
775  # Add to the exclude list unless we are at the last spatial param,
776  # in which case it is no longer needed.
777  # Hopefully this will save meaningful memory for large neighbour sets
778  if ($i != $#$spatial_params_ref) {
779  push @exclude, @{$nbr_list[$i]};
780  }
781  }
782  $self->add_to_lists (
783  element => $element,
784  $nbr_list_name => $nbr_list[$i],
785  );
786  }
787  }
788 
789  return wantarray ? @nbr_list : \@nbr_list;

1130 
1131 Biodiverse::Spatial - a set of spatial functions for a
1132 Biodiverse::BaseStruct object. POD IS MASSIVELY OUT OF DATE

1135

1138 =head1 DESCRIPTION
1139 
1140 These functions should be inherited by higher level objects through their @ISA
1141 list.
1142 
1143 MANY OF THESE HAVE BEEN MOVED BACK TO Biodiverse::BaseData.
1144 
1145 =head2 Assumptions
1146 
1147 Assumes C<Biodiverse::Common> is in the @ISA list.
1148 
1149 Almost all methods in the Biodiverse library use {keyword => value} pairs as a policy.
1150 This means some of the methods may appear to contain unnecessary arguments,
1151 but it makes everything else more consistent.
1152 
1153 List methods return a list in list context, and a reference to that list
1154 in scalar context.

1157 
1158 These assume you have declared an object called $self of a type that
1159 inherits these methods, normally:

1223 =item $self->get_distances (coord_array1 => $element1, coord_array2 => $element2);
1224 
1225 ALL THIS IS OUT OF DATE.
1226 
1227 Calculate the distances between the coords in two sets of elements using
1228 parameters derived from C<Biodiverse::Common::parse_spatial_params>.
1229 
1230 As of version 1 we only use Euclidean distance
1231 denoted by $D, $D[0], $D[1], $d[0], $d[1] etc. The actual values are
1232 determined using C<Biodiverse::Spatial::get_distances>.
1233 
1234 $D is the absolute euclidean distance across all dimensions.
1235 
1236 $D[0], $D[1] and so forth are the absolute distance in dimension 0, 1 etc.
1237 In most cases this $D[0] will be the X dimension, $D[1] will be the y dimension.
1238 
1239 $d[0], $d[1] and so forth are the signed distance in dimension 0, 1 etc.
1240 This allows us to extract all groups within some distance in some direction.
1241 As with standard cartesion plots, negative values are to the left or below (west or south),
1242 positive values to the right or above (east or north).
1243 As with $D[0], $d[0] will normally be the X dimension,
1244 $d[1] will be the y dimension.
1245 
1246 =item $sp->sp_calc(calculations => \%calculations);
1247 
1248 Calculate one or more spatial indices specified in %calculations using
1249 neighbourhood parameters stored in the objects's parameters.
1250 
1251 The results are stored in spatial object $sp.
1252 
1253 %calculations must have the same structure as that returned by
1254 C<Biodiverse::Indices::get_calculations>.
1255 
1256 Runs all available calculations if none are specified.
1257 
1258 Any other arguments are passed straight through to the indices.
1259 
1260 The C<cache_*> options allow the user to cache the element, label and ABC lists
1261 for direct export,
1262 although we will be adding methods to do this to save on storage space when it
1263 is exported (perl stores hash keys in a global list, so there is
1264 little overhead when using hash keys multiple times).
1265 
1266 The ABC lists are stored by default, as it is useful to display them and all
1267 the indices depend on them.
1268 
1269 Scalar results are added to the Spatial Output object's SPATIAL_OUTPUT hash.
1270 Any lists are added as separate lists in the object, rather than pollute
1271 the SPATIAL_OUTPUT hash with additional lists.
