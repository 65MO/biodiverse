180 # Processes keyboard shortcuts like CTRL-G = Go!
181 sub hotkeyHandler {
182  my ($widget, $event, $self) = @_;
183  my $retval;
184 
185  # stop recursion into onRun if shortcut triggered during processing
186  # (this happens because progress-dialogs pump events..)
187 
188  return 1 if ($handler_entered == 1);
189 
190  $handler_entered = 1;
191 
192  if ($event->type eq 'key-press') {
193  # if CTL- key is pressed
194  if ($event->state >= ['control-mask']) {
195  my $keyval = $event->keyval;
196  #print $keyval . "\n";
197 
198  # Go!
199  if ((uc chr $keyval) eq 'G') {
200  $self->onRun();
201  $retval = 1; # stop processing
202  }
203 
204  # Close tab (CTRL-W)
205  elsif ((uc chr $keyval) eq 'W') {
206  if ($self->getRemovable) {
207  $self->{gui}->removeTab($self);
208  $retval = 1; # stop processing
209  }
210  }
211 
212  # Change to next tab
213  elsif ($keyval eq Gtk2::Gdk->keyval_from_name ('Tab')) {
214  # switch tabs
215  #print "keyval is $keyval (tab), state is " . $event->state . "\n";
216  my $page_index = $self->getPageIndex;
217  $self->{gui}->switchTab (undef, $page_index + 1); # go right
218  }
219  elsif ($keyval eq Gtk2::Gdk->keyval_from_name ('ISO_Left_Tab')) {
220  # switch tabs
221  #print "keyval is $keyval (left tab), state is " . $event->state . "\n";
222  my $page_index = $self->getPageIndex;
223  $self->{gui}->switchTab (undef, $page_index - 1); # go left
224  }
225  }
226  }
227 
228  $handler_entered = 0;
229  $retval = 0; # continue processing
230  return $retval;
231 }
