64 sub calc_matrix_stats {
65  my $self = shift;
66  my %ABC = @_; # rest of args into a hash
67 
68 
69  my $matrix = $ABC{matrix_ref};
70 
71  # this will contain label_hash2 if element_list2 or label_list2 was specified
72  my %label_list = %{$ABC{label_hash_all}};
73 
74  # we need to get the distance between two percentiles
75  my @valueList;
76  my %done;
77 
78  my $labels_in_matrix = $matrix->get_elements;
79  my %tmp = %label_list;
80  delete @tmp{keys %$labels_in_matrix}; # get a list of those not in the matrix
81  delete @label_list{keys %tmp}; # these should be the ones in the matrix
82 
83 
84  foreach my $label1 (keys %label_list) {
85  foreach my $label2 (keys %label_list) {
86  next if $done{$label2};
87 
88  my $value = $matrix->get_value (element1 => $label1, element2 => $label2);
89  next if !defined $value;
90 
91  #$value = 0 if ! defined $value && $label1 eq $label2;
92 
93  push (@valueList, $value);
94  }
95  $done{$label1}++;
96  }
97 
98  my $stats = $stats_class->new;
99  $stats->add_data (\@valueList);
100 
101  $stats->sort_data;
102  my $valuesRef = [$stats->get_data];
103  my $null;
104  (my $mx_pct95, $null) = eval {$stats->percentile(95)};
105  (my $mx_pct05, $null) = eval {$stats->percentile( 5)};
106  (my $mx_pct25, $null) = eval {$stats->percentile(25)};
107  (my $mx_pct75, $null) = eval {$stats->percentile(75)};
108 
109  #print $stats->count . "\t";
110 
111  my %results = (
112  MX_MEAN => $stats->mean,
113  MX_SD => $stats->standard_deviation,
114  MX_N => $stats->count,
115  MX_MEDIAN => $stats->median,
116  MX_RANGE => $stats->sample_range,
117  MX_MINVALUE => $stats->min,
118  MX_MAXVALUE => $stats->max,
119  MX_SKEW => $stats->skewness,
120  MX_KURT => $stats->kurtosis,
121  MX_PCT95 => $mx_pct95,
122  MX_PCT05 => $mx_pct05,
123  MX_PCT25 => $mx_pct25,
124  MX_PCT75 => $mx_pct75,
125  MX_VALUES => $valuesRef,
126  MX_LABELS => \%label_list,
127  );
128 
129  return wantarray
130  ? %results
131  : \%results;

180 sub calc_compare_dissim_matrix_values {
181  my $self = shift;
182  my %args = @_;
183 
184  my $self_similarity = $args{self_similarity} || 0;
185 
186  my $label_list1 = $args{label_hash1};
187  my $label_list2 = $args{label_hash2};
188 
189  my $matrix = $args{matrix_ref};
190 
191  my (%tmp, %tmp2);
192  # delete elements from label_list1 that are not in the matrix
193  my $labels_in_matrix = $matrix -> get_elements;
194 
195  $label_list1 = $self -> get_list_intersection (
196  list1 => [keys %$labels_in_matrix],
197  list2 => [keys %$label_list1],
198  );
199 
200  $label_list2 = $self -> get_list_intersection (
201  list1 => [keys %$labels_in_matrix],
202  list2 => [keys %$label_list2],
203  );
204 
205  # we need to get the distance between and across two groups
206  my ($sumX, $sumXsqr, $count) = (undef, undef, 0);
207  #my ($totalSumX, $totalSumXsqr, $totalCount) = (undef, undef, 0);
208  my (%list1_hash, %list2_hash);
209 
210  #my (%done, %compared, %centre_compared);
211  BY_LABEL1:
212  foreach my $label1 (@{$label_list1}) {
213  $list1_hash{$label1} ++; # track the times it is used
214 
215  BY_LABEL2:
216  foreach my $label2 (@{$label_list2}) {
217 
218  #next BY_LABEL2 if $done{$label2}; # we've already compared these
219  $list2_hash{$label2} ++;
220 
221  my $value = $matrix->get_value(
222  element1 => $label1,
223  element2 => $label2
224  );
225 
226  # trap self-self values not in matrix but don't override ones that are
227  if (! defined $value) {
228  $value = $self_similarity;
229  }
230  elsif ($label1 eq $label2) {
231  $value = $self_similarity;
232  }
233

237  $count ++;
