NAME

Biodiverse::Indices::LabelProperties

SYNOPSIS

DESCRIPTION

Label property indices for the Biodiverse system. It is inherited by Biodiverse::Indices and not to be used on it own.

See http://code.google.com/p/biodiverse/wiki/Indices for more details.

sub calc_lbprop_hashes_abc2 {
my $self = shift;
my %args = @_;
# just grab the hash from the precalc results
my %objects = %{$args{LBPROP_STATS_OBJECTS_ABC2}};
my %results;
while (my ($prop, $stats_object) = each %objects) {
my @data = $stats_object->get_data();
my $key = $prop;
$key =~ s/DATA$/HASH2/;
foreach my $value (@data) {
$results{$key}{$value} ++;
}
}
return wantarray ? %results : \%results;
}

sub calc_lbprop_stats_abc2 {

while (my ($prop, $stats_object) = each %objects) {

sub get_metadata_calc_lbprop_hashes_abc2 {
my $self = shift;
my $desc = 'Hashes of the labels and their property values '
. 'used in the label properties calculations. '
. 'Hash keys are the property values, '
. 'hash values are the property value frequencies.';
my %indices;
my %prop_hash_names = $self->_get_lbprop_stats_hash_keynames;
while (my ($prop, $list_name) = each %prop_hash_names) {
$list_name =~ s/DATA$/HASH2/;
$indices{$list_name} = {
$list_name => 'Hash of values for property ' . $prop,
type => 'list',
};
}
my %arguments = (
description => $desc,
name => 'Label property hashes',
type => 'Element Properties',
pre_calc => ['get_lbp_stats_objects_abc2'],
uses_nbr_lists => 1,
indices => \%indices,
);
return wantarray ? %arguments : \%arguments;
}
